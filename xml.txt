<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<mapper namespace="dev.spring.petclinic.createowner.mapper.OwnerMapper">

    <!-- Owner 등록 -->
    <insert id="insertOwner" parameterType="dev.spring.petclinic.createowner.entity.Owner" useGeneratedKeys="true" keyProperty="id">
        INSERT INTO owners (first_name, last_name, address, city, telephone)
        VALUES (#{firstName}, #{lastName}, #{address}, #{city}, #{telephone})
    </insert>

    <!-- Owner 조회 by ID -->
    <select id="findById" parameterType="int" resultType="dev.spring.petclinic.createowner.entity.Owner">
        SELECT id, first_name, last_name, address, city, telephone
        FROM owners
        WHERE id = #{id}
    </select>

    <!-- 전화번호 중복 여부 확인 -->
    <select id="existsByPhoneNumber" parameterType="string" resultType="boolean">
        SELECT COUNT(*) > 0
        FROM owners
        WHERE telephone = #{telephone}
    </select>

    <!-- 추가: 전화번호로 Owner 조회 -->
    <select id="findByTelephone" parameterType="string" resultType="dev.spring.petclinic.createowner.entity.Owner">
        SELECT id, first_name, last_name, address, city, telephone
        FROM owners
        WHERE telephone = #{telephone}
    </select>

    <!-- Owner 수정 -->
    <update id="updateOwnerById" parameterType="dev.spring.petclinic.createowner.entity.Owner">
        UPDATE owners
        <set>
            <if test="firstName != null">first_name = #{firstName},</if>
            <if test="lastName != null">last_name = #{lastName},</if>
            <if test="address != null">address = #{address},</if>
            <if test="city != null">city = #{city},</if>
            <if test="telephone != null">telephone = #{telephone},</if>
        </set>
        WHERE id = #{id}
    </update>
</mapper>

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<mapper namespace="dev.spring.petclinic.findowner.mapper.OwnersMapper">

    <!-- 특정 ID로 Owner 조회 -->

    <select id="findOwnerById" parameterType="long" resultType="dev.spring.petclinic.findowner.model.Owners">
        SELECT id, first_name, last_name, address, city, telephone
        FROM owners
        WHERE id = #{id}
    </select>

    <resultMap id="ownerResultMap" type="dev.spring.petclinic.findowner.model.Owners">
        <id column="id" property="id" />
        <result column="first_name" property="firstName" />
        <result column="last_name" property="lastName" />
        <result column="address" property="address" />
        <result column="city" property="city" />
        <result column="telephone" property="telephone" />
        <!-- owner.id 값을 이용해서 pets를 조회 -->
        <collection property="pets" ofType="dev.spring.petclinic.findowner.model.Pet"
                    select="findPetsByOwnerId" column="id"/>
    </resultMap>

    <!-- 모든 Owner 목록 조회 -->
    <select id="findAllOwners" resultMap="ownerResultMap">
        SELECT id, first_name, last_name, address, city, telephone
        FROM owners
    </select>

    <!-- 성(last_name)으로 Owner 검색 -->
    <select id="findOwnersByLastName" parameterType="string" resultType="dev.spring.petclinic.findowner.model.Owners">
        SELECT id, first_name, last_name, address, city, telephone
        FROM owners
        WHERE last_name LIKE CONCAT('%', #{lastName}, '%')
    </select>

    <!-- Owner ID로 해당하는 Pet 정보 조회 -->
    <select id="findPetsByOwnerId" parameterType="long" resultType="dev.spring.petclinic.findowner.model.Pet">
        SELECT id, owner_id, name
        FROM pets
        WHERE owner_id = #{ownerId}
    </select>
</mapper>
